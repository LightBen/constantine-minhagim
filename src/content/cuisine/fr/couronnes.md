<p><strong>Ingr&eacute;dients</strong></p>

<ul>
	<li>250mL d&rsquo;eau</li>
	<li>250mL d&rsquo;huile</li>
	<li>300g de sucre</li>
	<li>3 sachets de sucre vanille</li>
	<li>3 sachets de levure chimique</li>
	<li>1 bouchon de fleur d&rsquo;oranger</li>
</ul>

<p><strong>Pr&eacute;paration</strong></p>

<ol>
	<li>M&eacute;langer le tout</li>
	<li>Ajouter 1kg de farine</li>
	<li>Laisser reposer 5mn</li>
	<li>Former les couronnes</li>
	<li>Cuire &agrave; 180C pendant 20 minutes</li>
</ol>
